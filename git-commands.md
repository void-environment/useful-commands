# Проверка статуса
```bash
git status                      # Текущее состояние репозитория
git status -s                   # Краткий формат статуса
```

# Просмотр истории
```bash
git log                         # Полная история коммитов
git log --oneline               # Компактная история
git log --graph --oneline       # Визуализация ветвления
git log --since="2024-01-01"    # Коммиты после указанной даты
```

# Просмотр изменений
```bash
git diff                        # Непроиндексированные изменения
git diff --staged               # Проиндексированные изменения
git diff HEAD                   # Все изменения относительно последнего коммита
```

# Отмена коммитов
```bash
git reset --soft HEAD~1         # Отменить коммит, сохранив изменения
git reset --mixed HEAD~1        # Отменить коммит и индексацию
git reset --hard HEAD~1         # Полностью удалить коммит и изменения
```

# Отмена изменений
```bash
git restore file.txt           # Отменить изменения в файле
git restore --staged file.txt  # Убрать файл из индекса
git checkout -- file.txt       # Альтернатива для отмены изменений
```

# Восстановление
```bash
git reflog                      # Показать историю всех действий
git cherry-pick <commit-hash>   # Применить чужой коммит к текущей ветке
```
